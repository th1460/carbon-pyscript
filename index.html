from pyodide.ffi import create_proxy, to_js import pandas as pd import js from pyodide.http import open_url def send_data(event): df = pd.read_csv(open_url(js.document.getElementById("input").value)) vars = df.columns.to_list() parent = js.document.getElementById("vars") for var in vars: item = js.document.createElement('bx-dropdown-item') item.textContent = var item.setAttribute('value', var.lower()) parent.appendChild(item) js.document.getElementById("button").addEventListener("click", create_proxy(send_data)) def get_analysis(event): df = pd.read_csv(open_url(js.document.getElementById("input").value)) df.columns = df.columns.str.lower() count = df[event.detail.item.value].value_counts() count = count.to_frame(name="value") count.reset_index(inplace=True) count.rename({'index': 'group'}, axis=1, inplace=True) count = count.to_dict(orient='records') objects = [] for i in count: objects.append(js.Object.fromEntries(to_js(i))) js.plot(objects) js.document.getElementById("vars").addEventListener("bx-dropdown-selected", create_proxy(get_analysis))
